package io.github.mcengine.api.artificialintelligence.dlc;

import org.bukkit.plugin.Plugin;

/**
 * Interface for AI DLC modules that can be dynamically loaded.
 */
public interface IMCEngineArtificialIntelligenceDLC {

    /**
     * Called when the DLC is loaded by the AI engine.
     *
     * @param plugin The plugin instance providing context.
     */
    void onLoad(Plugin plugin);
}
package io.github.mcengine.dlc.sample;

import io.github.mcengine.api.artificialintelligence.dlc.IMCEngineArtificialIntelligenceDLC;
import org.bukkit.plugin.Plugin;

/**
 * Example implementation of an AI DLC module.
 */
public class SampleAIDLC implements IMCEngineArtificialIntelligenceDLC {

    @Override
    public void onLoad(Plugin plugin) {
        plugin.getLogger().info("[SampleAIDLC] DLC Loaded!");
    }
}
